### Key Snippets

**Component (Signal + standalone):**
```typescript
@Component({
  standalone: true,
  selector: 'app-transaction-list',
  templateUrl: './transaction-list.component.html',
  imports: [CommonModule, ReactiveFormsModule, MatPaginatorModule, TransactionFilterComponent]
})
export class TransactionListComponent {
  private service = inject(TransactionService);

  filter = signal<TransactionFilter>({ from: thirtyDaysAgo(), to: today(), type: 'ALL' });
  page = signal({ index: 0, size: 20 });

  result = computed(() => this.service.query(this.filter(), this.page()));
  // ↑ Uses resource() / rxResource() internally for loading/error states

  onFilterChange(f: TransactionFilter) { this.filter.set(f); this.page.set({ index: 0, size: 20 }); }
  onPageChange(e: PageEvent) { this.page.set({ index: e.pageIndex, size: e.pageSize }); }

  trackByTxnId = (_: number, t: Transaction) => t.transactionId;  // ← required by reviewer
}
```

**Template (states + a11y):**
```html
@if (result().isLoading) {
  <app-transaction-skeleton aria-label="Loading transactions"/>
}
@else if (result().error()) {
  <app-error-state [retry]="onRetry" aria-live="polite"/>
}
@else if (result().value()?.content?.length === 0) {
  <app-empty-state message="{{ 'transactions.empty' | translate }}"/>
}
@else {
  <ul role="list">
    @for (txn of result().value()!.content; track trackByTxnId($index, txn)) {
      <li>
        <span aria-label="Date">{{ txn.executedAt | date }}</span>
        <span aria-label="Amount">{{ txn.amount | currency:txn.currency }}</span>
        <span aria-label="Counterparty">{{ txn.counterpartyName }} ({{ txn.counterpartyAccountMasked }})</span>
        <span [class]="'status-' + txn.status.toLowerCase()">{{ txn.status }}</span>
      </li>
    }
  </ul>
  <mat-paginator
    [length]="result().value()!.totalElements"
    [pageSize]="page().size"
    [pageIndex]="page().index"
    (page)="onPageChange($event)"
    aria-label="Pagination"/>
}
```

**Service (typed from OpenAPI):**
```typescript
@Injectable({ providedIn: 'root' })
export class TransactionService {
  private api = inject(TransactionApi);  // generated by openapi-generator
  query = rxResource({
    request: () => ({ filter: this.filterSignal(), page: this.pageSignal() }),
    loader: ({ request }) => this.api.getTransactions(request.filter, request.page)
  });
}
```

## Talking points (60 sec)

1. "Standalone component + Signal + computed — no NgModule, no manual subscribe"
2. "5 states ในเทมเพลตเดียว: loading, error, empty, data, paginator — ตามที่ Designer ระบุ"
3. "trackBy ใส่จากเริ่ม — เพื่อให้ scroll position + animation ไม่กระตุก"
4. "aria-label ทุก data cell — WCAG 2.1 AA compliant"
5. "i18n keys — ไม่มี hardcoded Thai/English ในเทมเพลต"
